##### 2,问题数据缓存

- 目的: 可以将问题数据缓存到redis中

- 操作流程:

  - 1, 问题工具类(common/cache/question.py)

    - ```python
      #1,获取question对象,缓存类
      class QuestionBasicCache(object):
      
          #1,初始化方法
          def __init__(self,id):
              self.id = id
      
          #2,获取对象
          def get(self):
      
              #1,获取redis中是否有缓存
              redis_cluster = current_app.redis_cluster
              key = "question_basic:{}".format(self.id)
              try:
                  redis_data = redis_cluster.get(key)
              except Exception as e:
                  current_app.logger.error(e)
                  redis_data = None
      
              #2,判断是否存在数据
              if redis_data:
                  return json.loads(redis_data)
      
              #3,获取对象数据
              obj = self.get_data_obj()
      
              #4,对象转字典
              data_dict = self.create_obj_dict(obj)
      
              #5,缓存数据
              try:
                  redis_cluster.setex(key,60*5,json.dumps(data_dict))
              except Exception as e:
                  current_app.logger.error(e)
      
              #6,返回
              return data_dict
      
      
          #获取对象数据
          def get_data_obj(self):
              # 1,根据id获取question对象
              obj = Question.query.options(load_only(Question.id,
                                                     Question.title,
                                                     Question.user_id,
                                                     Question.expertise_id,
                                                     Question.city_id,
                                                     Question.ctime
                                                     )).filter_by(id=self.id).first()
      
              # 2,将question对象,转换成字典
              if not obj:
                  return None
      
              return obj
      
          #将对象转成字典数据
          def create_obj_dict(self,obj):
              # 获取城市
              city = City.query.filter_by(id=obj.city_id).first()
      
              author = User.query.filter(User.id == obj.user_id).first()
              obj_dict = {
                  "id": obj.id,
                  "title": obj.title,
                  "author_name": author.name,
                  "expertise": EXPERTISE[obj.expertise_id - 1][1] if obj.expertise_id else None,
                  "city": city.name if city else None,
                  "create_time": obj.ctime.strftime("%Y-%m-%d %H:%M%S"),
                  "user_id": author.id
              }
      
              return obj_dict
      
      ```

      class SearchResource(Resource):
          def get(self):
      
              #1,解析参数
              rp = RequestParser()
              rp.add_argument("k",type=str,required=True,location="args")
              rp.add_argument("cur_page",type=int,required=True,location="args")
              rp.add_argument("per_page_count",type=int,required=True,location="args")
      
              #2,获取参数
              args = rp.parse_args()
              k = args.k
              cur_page = args.cur_page
              per_page_count = args.per_page_count
      
              #3,查询数据
              query = {
                  'from':(cur_page - 1)*per_page_count,
                  'size':per_page_count,
                  'query':{
                      'bool':{
                          'must':[{'match':{'_all':k}}],
                          'filter':[{'term':{'status':'2'}}]
                      }
                  }
              }
              ret = current_app.es.search(index="qusts",doc_type="qust",body=query)
      
              #4,拼接数据
              # print(ret)
              hits_list = ret['hits']['hits']
              id_list = []
              dataList = []
              for hit in hits_list:
                  id_list.append(hit["_id"])
      
              for id in id_list:
                  question = QuestionBasicCache(id).get()
                  user = UserCache(question['user_id']).get()
      
                  data_dict = {
                      "city":question["city"],
                      "create_time":question["create_time"],
                      "answer_count":0,
                      "id":question["id"],
                      "author_name":user["name"],
                      "title":question["title"],
                      "expertise":question["expertise"],
                      "author_photo_url":user["profile_photo"],
                  }
      
                  dataList.append(data_dict)
      
              #5,返回响应
              return dataList
          
      ```
